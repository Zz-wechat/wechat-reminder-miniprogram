# 微信小程序开发快速入门

## 📝 基础示例

### 1. 简单的页面结构
```html
<!-- pages/demo/index.wxml -->
<view class="container">
  <text>{{message}}</text>
  <button bindtap="changeText">点击改变文字</button>
</view>
```

```css
/* pages/demo/index.wxss */
.container {
  padding: 20rpx;
  text-align: center;
}

button {
  margin-top: 40rpx;
  background-color: #07c160;
  color: white;
}
```

```javascript
// pages/demo/index.js
Page({
  data: {
    message: 'Hello World!'
  },
  
  changeText() {
    this.setData({
      message: '你点击了按钮！'
    })
  }
})
```

### 2. 数据绑定和列表渲染
```html
<!-- 显示数组数据 -->
<view wx:for="{{items}}" wx:key="index">
  <text>{{index + 1}}. {{item}}</text>
</view>

<!-- 条件渲染 -->
<view wx:if="{{show}}">显示内容</view>
<view wx:else>隐藏内容</view>
```

### 3. 事件处理
```javascript
// 页面事件
onLoad() {
  console.log('页面加载')

onShow() {
  console.log('页面显示')

  console.log('页面隐藏')
},

// 自定义事件
handleTap(e) {
  console.log('点击事件', e.currentTarget.dataset)

// 表单事件
handleInput(e) {
  console.log('输入内容:', e.detail.value)
}
```

## 🔧 常用API示例

### 1. 本地存储
```javascript
// 存储数据
wx.setStorageSync('key', 'value')

// 读取数据
const value = wx.getStorageSync('key')

// 删除数据
wx.removeStorageSync('key')
```

### 2. 网络请求
```javascript
wx.request({
  url: 'https://api.example.com/data',
  method: 'GET',
  success(res) {
    console.log('请求成功:', res.data)
  },
  fail(error) {
    console.log('请求失败:', error)
  }
})
```

### 3. 用户交互
```javascript
// 显示提示
wx.showToast({
  title: '操作成功',
  icon: 'success',
  duration: 2000
})

// 显示模态框
wx.showModal({
  title: '确认删除',
  content: '确定要删除这个提醒吗？',
  success(res) {
    if (res.confirm) {
      console.log('用户点击确定')
    }
  }

// 显示加载中
wx.showLoading({
  title: '加载中...'
})
setTimeout(() => {
  wx.hideLoading()
}, 2000)
```

## 🎯 提醒小程序核心功能示例

### 1. 创建提醒
```javascript
// 创建提醒对象
const reminder = {
  id: Date.now(),
  content: '下午3点开会',
  time: '15:00',
  date: '2024-03-02',
  repeat: 'none', // none, daily, weekly, monthly
  enabled: true,
  createdAt: new Date().toISOString()
}

// 保存到本地
let reminders = wx.getStorageSync('reminders') || []
reminders.push(reminder)
wx.setStorageSync('reminders', reminders)
```

### 2. 获取今日提醒
```javascript
function getTodayReminders() {
  const allReminders = wx.getStorageSync('reminders') || []
  const today = new Date().toISOString().split('T')[0]
  
  return allReminders.filter(reminder => {
    return reminder.date === today && reminder.enabled
  }).sort((a, b) => {
    // 按时间排序
    return a.time.localeCompare(b.time)
  })
}
```

### 3. 删除提醒
```javascript
function deleteReminder(id) {
  let reminders = wx.getStorageSync('reminders') || []
  reminders = reminders.filter(reminder => reminder.id !== id)
  wx.setStorageSync('reminders', reminders)
  
  wx.showToast({
    title: '删除成功',
    icon: 'success'
  })
}
```

## 📱 页面跳转和导航

### 1. 页面路由
```javascript
// 跳转到新页面
wx.navigateTo({
  url: '/pages/reminder/create'
})

// 返回上一页
wx.navigateBack()

// 重定向页面（关闭当前页）
wx.redirectTo({
  url: '/pages/index/index'
})

// 切换到Tab页
wx.switchTab({
  url: '/pages/index/index'
})
```

### 2. 页面传参
```javascript
// 发送参数
wx.navigateTo({
  url: '/pages/reminder/detail?id=123&type=daily'
})

// 接收参数
Page({
  onLoad(options) {
    console.log('页面参数:', options.id) // 123
    console.log('页面参数:', options.type) // daily
  }
})
```
